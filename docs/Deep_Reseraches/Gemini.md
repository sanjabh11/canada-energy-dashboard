        self.model = AutoModelForSeq2SeqLM.from_pretrained(model_id)
        self.device = torch.device("cuda" if torch.cuda.is_available() else "cpu")
        self.model.to(self.device)
        self.model.eval()

    def generate_zero_shot_forecast(self, history_tensor, prediction_length=24):
        """
        Executes non-autoregressive quantile prediction.
        """
        history = torch.tensor(history_tensor, dtype=torch.float32).to(self.device)
        
        # Scaling input (mean scaling)
        mean = history.mean(dim=-1, keepdim=True)
        scale = torch.clamp(history.std(dim=-1, keepdim=True), min=1e-5)
        scaled_history = (history - mean) / scale
        
        with torch.no_grad():
            outputs = self.model.generate(
                scaled_history,
                max_length=prediction_length,
                num_beams=1,
                use_cache=True
            )
            # Rescale output to physical scale
            forecasts = (outputs * scale.unsqueeze(-1)) + mean.unsqueeze(-1)
        return forecasts.cpu().numpy()

import torch.nn as nn

class SeriesDecomposition(nn.Module):
    """
    Decomposes input series into Trend and Seasonal components using moving averages.
    """
    def __init__(self, kernel_size):
        super(SeriesDecomposition, self).__init__()
        self.moving_avg = nn.AvgPool1d(kernel_size=kernel_size, stride=1, padding=kernel_size // 2)

    def forward(self, x):
        x_permuted = x.permute(0, 2, 1)
        trend = self.moving_avg(x_permuted).permute(0, 2, 1)
        seasonal = x - trend
        return seasonal, trend

class PowerMambaFusionBlock(nn.Module):
    """
    Fuses weather covariates into load channels using pre-decomposition summation.
    """
    def __init__(self, d_model, num_variables, kernel_size=25):
        super(PowerMambaFusionBlock, self).__init__()
        self.decomposition = SeriesDecomposition(kernel_size)
        
        # Feature embeddings
        self.value_embedding = nn.Linear(num_variables, d_model)
        self.weather_embedding = nn.Linear(num_variables, d_model)
        
        # Linear layer for combining streams
        self.fuse_layer = nn.Linear(d_model * 2, d_model)
        
    def forward(self, historical_load, weather_covariates):
        """
        Fuses features and projects them into seasonal and trend components.
        """
        load_emb = self.value_embedding(historical_load)
        weather_emb = self.weather_embedding(weather_covariates)
        
        fused = self.fuse_layer(torch.cat([load_emb, weather_emb], dim=-1))
        
        # Decompose fused representation
        seasonal_fused, trend_fused = self.decomposition(fused)
        
        return seasonal_fused, trend_fused

class AdaptiveConformalEngineWithReset:
    """
    ACI engine with daily reset optimized for solar PV and prosumer flexibility.
    """
    def __init__(self, target_coverage=0.90, lr=0.005):
        self.alpha = 1.0 - target_coverage # Target error rate (e.g., 0.10)
        self.lr = lr
        self.alpha_t = self.alpha # Running miscoverage parameter
        self.nonconformity_scores = []

    def calibrate_step(self, predicted_lower, predicted_upper, true_val, hour_of_day):
        """
        Updates the miscoverage parameter based on coverage errors,
        resetting at dawn (hour 5) to mitigate PV nighttime bias.
        """
        if hour_of_day == 5:
            self.alpha_t = self.alpha # Reset to baseline
            
        # Determine nonconformity score
        n_score = max(predicted_lower - true_val, true_val - predicted_upper)
        self.nonconformity_scores.append(n_score)
        
        # Determine coverage error (1 if actual is outside predicted interval, else 0)
        coverage_error = 1.0 if n_score > 0 else 0.0
        
        # Adapt running parameter
        self.alpha_t = self.alpha_t + self.lr * (coverage_error - self.alpha)
        self.alpha_t = np.clip(self.alpha_t, 0.01, 0.99)
        
        return self.alpha_t

    def get_conformal_interval(self, pred_lower, pred_upper):
        """
        Applies calibrated scaling to new raw interval boundaries.
        """
        if len(self.nonconformity_scores) < 20:
            return pred_lower, pred_upper
            
        q_level = 1.0 - self.alpha_t
        q_correction = np.percentile(self.nonconformity_scores, q_level * 100.0)
        
        conformal_lower = pred_lower - q_correction
        conformal_upper = pred_upper + q_correction
        
        return conformal_lower, conformal_upper

class ReducedSTGNNFaultLocator(nn.Module):
    """
    Spatio-Temporal GNN built on measured-only topology.
    Filters out unobserved nodes to achieve a 6x speedup.
    """
    def __init__(self, in_channels, hidden_channels, out_channels=1):
        super(ReducedSTGNNFaultLocator, self).__init__()
        self.conv1 = SAGEConv(in_channels, hidden_channels)
        self.conv2 = SAGEConv(hidden_channels, hidden_channels)
        
        self.temporal_gru = nn.GRU(hidden_channels, hidden_channels, batch_first=True)
        self.predictor = nn.Linear(hidden_channels, out_channels)
        
    def forward(self, x, edge_index, edge_weight=None):
        """
        x shape: [Batch, Measured_Nodes, Timesteps, Input_Features]
        """
        batch_size, num_nodes, timesteps, num_features = x.shape
        
        # Flatten batch and spatial dimensions for topological processing
        x_reshaped = x.permute(0, 2, 1, 3).reshape(batch_size * timesteps, num_nodes, num_features)
        
        # Apply GNN layers sequentially across graph snapshots
        gnn_out = []
        for t in range(timesteps):
            xt = x[:, :, t, :]
            temp_list = []
            for b in range(batch_size):
                h = torch.relu(self.conv1(xt[b], edge_index))
                h = torch.relu(self.conv2(h, edge_index))
                temp_list.append(h)
            gnn_out.append(torch.stack(temp_list))
        
        # Reshape to [Batch, Measured_Nodes, Timesteps, Hidden_Channels]
        spatial_features = torch.stack(gnn_out, dim=2)
        
        # Temporal processing on each node independently
        temporal_out = []
        for n in range(num_nodes):
            node_seq = spatial_features[:, n, :, :]
            _, h_n = self.temporal_gru(node_seq)
            temporal_out.append(h_n.squeeze(0))
            
        node_features = torch.stack(temporal_out, dim=1)
        
        logits = self.predictor(node_features).squeeze(-1)
        return torch.sigmoid(logits)
